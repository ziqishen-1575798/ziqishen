Y = 16.6
X = 5
num_x3 = np.linspace(0, 10, 1000)
sig = 2
val = stats.norm.pdf(num_x3, X, sig)
plt.plot(num_x3, val)
plt.xlabel('Number of Outcomes')
plt.ylabel('Probability Density')
signal = 1 - stats.norm.cdf(Y, X, sig)
sigma = stats.norm.ppf(1-signal)
print('Sigma = ' , sigma)
# As our calculation has shown that the significance of our detection is 5.799, which is greater than the 5 sigma. Therefore, we can claim a discovery.
